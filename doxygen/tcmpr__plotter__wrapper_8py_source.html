164  util.mkdir_p(self.output_base_dir)
165 
166  # If input data is a file, create a single command and invoke R script.
167  if os.path.isfile(self.input_data):
168  self.logger.debug("Currently plotting " + self.input_data)
169  cmds_dict[' -lookin '] = self.input_data
170 
171  # Special treatment of the "optional" output_base_dir option
172  # because we are supporting the plotting of multiple tcst files
173  # in a directory.
174  if self.output_base_dir:
175  # dated_output_dir = self.create_output_subdir(self.input_data)
176  cmds_dict[' -outdir '] = self.output_base_dir
177 
178  # Generate the list, where the -args are separated by their
179  # values.
180  full_cmd_list = ['Rscript' + self.tcmpr_script]
181  for key, value in cmds_dict.items():
182  full_cmd_list.append(key)
183  full_cmd_list.append(value)
184 
185  # Separate the 'Rscript' portion from the args, to conform to
186  # produtil's exe syntax.
187  cmd = exe(full_cmd_list[0])[full_cmd_list[1:]] > '/dev/null'
188  self.logger.debug("Command run " +
189  cmd.to_shell())
190  self.logger.info("Generating requested plots for " +
191  self.input_data)
192  # pylint:disable=unnecessary-pass
193  # If a tc file is empty, continue to the next, thus the pass
194  # isn't unnecessary.
195  try:
196  checkrun(cmd)
197  except produtil.run.ExitStatusException as ese:
198  self.logger.warn("plot_tcmpr.R returned non-zero"
199  " exit status, "
200  "tcst file may be missing data, "
201  "continuing: " + repr(ese))
202 
203  # If the input data is a directory, create a list of all the
204  # files in the directory and invoke the R script for this list
205  # of files.
206  elif os.path.isdir(self.input_data):
207  self.logger.debug("plot all files in directory " +
208  self.input_data)
209  cmds_dict = self.retrieve_optionals()
210  all_tcst_files_list = util.get_files(self.input_data, ".*.tcst",
211  self.logger)
212  all_tcst_files = ' '.join(all_tcst_files_list)
213  self.logger.debug("num of files " + str(len(all_tcst_files)))
214  # Append the mandatory -lookin option to the base command.
215  cmds_dict['-lookin'] = all_tcst_files
216  if self.output_base_dir:
217  cmds_dict['-outdir'] = self.output_base_dir
218  self.logger.debug("Creating dated output dir " +
219  self.output_base_dir)
220 
221  # Create the full_cmd_list from the keys and values of the
222  # cmds_dict and then form one command list.
223  full_cmd_list = list()
224  full_cmd_list.append("Rscript")
225  full_cmd_list.append(self.tcmpr_script)
226  for key, value in cmds_dict.items():
227  full_cmd_list.append(key)
228  if key == '-lookin':
229  # treat the list of dirs in -lookin differently,
230  # append each individual directory to replicate original
231  # implementation's behavior of splitting the commands
232  # by whitespace and assigning each command to an item
233  # in a list.
234  for tcst_file in all_tcst_files_list:
235  full_cmd_list.append(tcst_file)
236  elif key == '-plot':
237  # plot types list is also appended as a single string,
238  # delimited by ','.
239  full_cmd_list.append(','.join(value))
240  elif key == '-dep':
241  # dependant variables list items are appended
242  # as one string. Convert list into a string delimited
243  # by ','.
244  full_cmd_list.append(','.join(value))
245 
246  else:
247  full_cmd_list.append(value)
248 
249  # Separate the 'Rscript' portion from the args, to conform to
250  # produtil's exe syntax.
251  cmd = exe(full_cmd_list[0])[full_cmd_list[1:]] > '/dev/null'
252 
253  # This can be a very long command if the user has
254  # indicated a directory. Only log this if necessary.
255  # self.logger.debug("DEBUG: Command run " + cmd.to_shell())
256  # cmd_str = ' '.join(full_cmd_list)
257  # cmd_list = 'Rscript ' + cmd_str
258  # self.logger.debug('TCMPR Command run: ' + cmd_str)
259 
260  # Now run the command via produtil
261  try:
262  checkrun(cmd)
263  except produtil.run.ExitStatusException as ese:
264  # If the tcst file is empty (with the exception of the
265  # header), or there is some other problem, then
266  # plot_tcmpr.R will return with a non-zero exit status of 1
267  self.log_error("plot_tcmpr.R returned non-zero"
268  " exit status, tcst file may be missing"
269  " data... continuing: " + str(ese))
270  sys.exit(1)
271  else:
272  self.log_error("Expected input is neither a file nor directory,"
273  "exiting...")
274  sys.exit(1)
275 
276  self.logger.info("Plotting complete")
