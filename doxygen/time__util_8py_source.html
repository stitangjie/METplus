28  Args:
29  @param value time value optionally ending in Y,m,d,H,M,S
30  Valid options match format 3600, 3600S, 60M, or 1H
31  @param default_unit unit to assume if no letter is found at end of value
32  @return relativedelta object containing offset time"""
33  if isinstance(value, int):
34  return get_relativedelta(str(value), default_unit)

38  match = re.match(reg, value)

43  unit_value = match.group(3)

102  """!Check relativedelta object contents and compute the total number of seconds
103  in the time. Return None if years or months are set, because the exact number
104  of seconds cannot be calculated without a relative time"""
105  # return None if input is not relativedelta object
106  if not isinstance(lead, relativedelta):
107  return None

109  # if valid time is specified, use it to determine seconds
110  if valid_time is not None:
111  return int((valid_time - (valid_time - lead)).total_seconds())
112 
113  if lead.months != 0 or lead.years != 0:
114  return None
