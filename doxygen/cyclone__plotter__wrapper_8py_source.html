98  self.logger.debug("Parsing file {}".format(init_file))
99 
100  # Extract information from the header, which is
101  # the first line.
102  header = infile.readline()
103  # print("header: {}".format(header))
104  column_indices = self.get_columns_and_indices(header)
105 
106  # For the remaining lines of this file,
107  # retrieve information from each row:
108  # lon, lat, init time, lead hour, valid time,
109  # model name, mslp, and basin.
110  # NOTE: Some of these aren't used until we fully
111  # emulate Guang Ping's plots.
112  for line in infile:
113  track_dict = {}
114  col = line.split()
115  lat = col[column_indices['ALAT']]
116  lon = col[column_indices['ALON']]
117  init_time = col[column_indices['INIT']]
118  fcst_lead_hh = \

127  # basin = col[column_indices['BASIN']]
128 
129  # Check for NA values in lon and lat, skip to
130  # next line in file if 'NA' is encountered.
131  if lon == 'NA' or lat == 'NA':
132  continue
133  else:
134  track_dict['lon'] = float(lon)
135  track_dict['lat'] = float(lat)
136 
137  # If the lead hour is 'NA', skip to next line.
138  # The track data was very likely generated with
139  # TC-Stat set to match-pairs set to True.
140  lead_hr = self.extract_lead_hr(fcst_lead_hh)
141  if lead_hr == 'NA':
142  continue
143 
144  # Check that the init date, init hour
145  # and model name are what the user requested.
146  init_ymd, init_hh = \
147  self.extract_date_and_time_from_init(init_time)
148 
149  if init_ymd == self.init_date and \
150  init_hh == self.init_hr:
151  if model_name == self.model:
152  # Check for the requested model,
153  # if the model matches the requested
154  # model name, then we have all the
155  # necessary information to continue.
156  # Store all data in dictionary
157  track_dict['fcst_lead_hh'] = fcst_lead_hh
158  track_dict['init_time'] = init_time
159  track_dict['model_name'] = model_name
160  track_dict['valid_time'] = valid_time
161  track_dict['storm_id'] = storm_id
162 
163  # Identify the 'first' point of the
164  # storm track. If the storm id is novel, then
165  # retrieve the date and hh from the valid time
166  if storm_id in self.unique_storm_id:
167  track_dict['first_point'] = False
168  track_dict['valid_dd'] = ''
169  track_dict['valid_hh'] = ''
170  else:
171  self.unique_storm_id.add(storm_id)
172  # Since this is the first storm_id with
173  # a valid value for lat and lon (ie not
174  # 'NA'), this is the first track point
175  # in the storm track and will be
176  # labelled with the corresponding
177  # date/hh z on the plot.
178  valid_match = \
179  re.match(r'[0-9]{6}([0-9]{2})_' +
180  '([0-9]{2})[0-9]{4}',
181  track_dict['valid_time'])
182  if valid_match:
183  valid_dd = valid_match.group(1)
184  valid_hh = valid_match.group(2)
185  else:
186  # Shouldn't get here if this is
187  # the first point of the track.
188  valid_dd = ''
189  valid_hh = ''
190  track_dict['first_point'] = True
191  track_dict['valid_dd'] = valid_dd
192  track_dict['valid_hh'] = valid_hh
193 
194  # Identify points based on valid time (hh).
195  # Useful for plotting later on.
196  valid_match = \
197  re.match(r'[0-9]{8}_([0-9]{2})[0-9]{4}',
198  track_dict['valid_time'])
199  if valid_match:
200  # Since we are only interested in 00,
201  # 06, 12, and 18 hr times...
202  valid_hh = valid_match.group(1)
203 
204  if valid_hh == '00' or valid_hh == '12':
205  track_dict['lead_group'] = '0'
206  elif valid_hh == '06' or valid_hh == '18':
207  track_dict['lead_group'] = '6'
208  else:
209  # To gracefully handle any hours other
210  # than 0, 6, 12, or 18
211  track_dict['lead_group'] = ''
212 
213  all_tracks_list.append(track_dict)
214 
215  # For future work, support for MSLP when
216  # generating regional plots-
217  # implementation goes here...
218  # Finishing up, do any cleaning up,
219  # logging, etc.
220  self.logger.info("All criteria met, " +
221  "saving track data init " +
222  track_dict['init_time'] +
223  " lead " +
224  track_dict['fcst_lead_hh'] +
225  " lon " +
226  str(track_dict['lon']) +
227  " lat " +
228  str(track_dict['lat']))
229 
230  else:
231  # Not the requested model, move to next
232  # row of data
233  continue
234 
235  else:
236  # Not the requested init ymd move to next
237  # row of data
238  continue
239 
240  # Now separate the data based on storm id.
241  for cur_unique in self.unique_storm_id:
242  cur_storm_list = []
243  for cur_line in all_tracks_list:
244  if cur_line['storm_id'] == cur_unique:
245  cur_storm_list.append(cur_line)
246  else:
247  # Continue to next line in all_tracks_list
248  continue
249 
250  # Create the storm_id_dict, which is the data
251  # structure used to separate the storm data based on
252  # storm id.
253  self.storm_id_dict[cur_unique] = cur_storm_list
254 
255  else:

277  @staticmethod
278  def extract_date_and_time_from_init(init_time_str):
279  """ Extract and return the YYYYMMDD portion and the
280  hh portion from the init time taken from the .tcst file.
281  """
282  match_ymd = re.match(r'([0-9]{8}).*', init_time_str)
283  match_hh = re.match(r'[0-9]{8}_([0-9]{2,3})[0-9]{4}', init_time_str)
284  # pylint:disable=no-else-return
285  # Explicitly return None if no match is found
286  if match_ymd and match_hh:
287  return match_ymd.group(1), match_hh.group(1)
288  else:
289  return None

291  @staticmethod
292  def extract_lead_hr(lead_str):
293  """ Extract and return the lead hours from the hhmmss lead string.
294  """
295  match = re.match(r'([0-9]{2,3})[0-9]{4}', str(lead_str))
296  if match:
297  return match.group(1)
