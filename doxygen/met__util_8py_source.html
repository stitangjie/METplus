133 def run_metplus(config, process_list):
134  total_errors = 0
135  try:
136  processes = []
137  for item in process_list:
138  try:
139  logger = config.log(item)
140  command_builder = \
141  getattr(sys.modules['__main__'],
142  item + "Wrapper")(config, logger)
143  # if Usage specified in PROCESS_LIST, print usage and exit
144  if item == 'Usage':
145  command_builder.run_all_times()
146  return 0
147  except AttributeError:
148  raise NameError("Process %s doesn't exist" % item)
149 
150  processes.append(command_builder)
151 
152  # check if all processes initialized correctly
153  allOK = True
154  for process in processes:
155  if not process.isOK:
156  allOK = False
157  class_name = process.__class__.__name__.replace('Wrapper', '')
158  logger.error("{} was not initialized properly".format(class_name))
159 
160  # exit if any wrappers did not initialized properly
161  if not allOK:
162  logger.info("Refer to ERROR messages above to resolve issues.")
163  return 1
164 
165  loop_order = config.getstr('config', 'LOOP_ORDER', '')
166 
167  if loop_order == "processes":
168  for process in processes:
169  process.run_all_times()
170 
171  elif loop_order == "times":
172  loop_over_times_and_call(config, processes)
173 
174  else:
175  logger.error("Invalid LOOP_ORDER defined. " + \
176  "Options are processes, times")
177  return 1
178 
179  # compute total number of errors that occurred and output results
180  for process in processes:
181  if process.errors != 0:
182  process_name = process.__class__.__name__.replace('Wrapper', '')
183  error_msg = '{} had {} error'.format(process_name, process.errors)
184  if process.errors > 1:
185  error_msg += 's'
186  error_msg += '.'
187  logger.error(error_msg)
188  total_errors += process.errors
189 
190  return total_errors
191  except:
192  logger.exception("Fatal error occurred")
193  logger.info(f"Check the log file for more information: {config.getstr('config', 'LOG_METPLUS')}")
194  return 1
