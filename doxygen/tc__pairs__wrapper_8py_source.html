457  def process_data(self, basin, cyclone, model_list, time_info):
458  """!Find requested files and run tc_pairs
459  Args:
460  @param basin region of storm from config
461  @param cyclone ID number of cyclone from config
462  @param model_list list of models that be available
463  @param time_info object containing timing information to process
464  """
465  # get bdeck file
466  bdeck_files = []
467 
468  # set regex expressions for basin and cyclone if wildcard is used
469  # cast cyclone value to integer if it is not a wildcard
470  if cyclone != '*':
471  cyclone_regex = cyclone
472  else:
473  cyclone_regex = "([0-9]{2,4})"
474 
475  if basin != '??':
476  basin_regex = basin
477  else:
478  basin_regex = "([a-zA-Z]{2})"
479 
480  # get search expression for bdeck files to pass to glob
481  string_sub = StringSub(self.logger,
482  self.c_dict['BDECK_TEMPLATE'],
483  basin=basin,
484  cyclone=cyclone,
485  **time_info)
486  bdeck_glob = os.path.join(self.c_dict['BDECK_DIR'],
487  string_sub.do_string_sub())
488  self.logger.debug('Looking for BDECK: {}'.format(bdeck_glob))
489 
490  # get all files that match expression
491  bdeck_files = sorted(glob.glob(bdeck_glob))
492 
493  # if no bdeck_files found
494  if len(bdeck_files) == 0:
495  template = self.c_dict['BDECK_TEMPLATE']
496  self.log_error(f'No BDECK files found searching for basin {basin} and '
497  f'cyclone {cyclone} using template {template}')
498  return False
499 
500  # find corresponding adeck or edeck files
501  for bdeck_file in bdeck_files:
502  self.logger.debug('Found BDECK: {}'.format(bdeck_file))
503 
504  # set current basin and cyclone from bdeck file
505  # if basin or cyclone are a wildcard, these will be
506  # replaced by the value pulled from the bdeck file
507  current_basin = basin
508  current_cyclone = cyclone
509 
510  # if wildcard was used in bdeck, pull out what was
511  # substituted for * to find corresponding bdeck file
512  matches = []
513  if '*' in bdeck_glob or '?' in bdeck_glob:
514  # get regex expression to pull out basin and cyclone
515  string_sub = StringSub(self.logger,
516  self.c_dict['BDECK_TEMPLATE'],
517  basin=basin_regex,
518  cyclone=cyclone_regex,
519  **time_info)
520  bdeck_regex = os.path.join(self.c_dict['BDECK_DIR'],
521  string_sub.do_string_sub())
522 
523  # capture wildcard values in template
524  bdeck_regex = bdeck_regex.replace('*', '(.*)')
525  bdeck_regex = bdeck_regex.replace('?', '(.)')
526 
527  match = re.match(bdeck_regex, bdeck_file)
528  if match:
529  matches = match.groups()
530  tags = get_tags(self.c_dict['BDECK_TEMPLATE'])
531  match_count = 0
532  for tag in tags:
533  # if wildcard is set for tag found, get value
534  # if wildcard if found in template, increment index
535  if tag == 'basin' and basin == '??':
536  current_basin = matches[match_count]
537  match_count += 1
538  elif tag == 'cyclone' and cyclone == '*':
539  current_cyclone = matches[match_count]
540  match_count += 1
541  elif tag == '*' or tag == '?':
542  match_count += 1
543 
544  # create lists for deck files, put bdeck in list so it can be handled
545  # the same as a and e for reformatting even though it will always be
546  # size 1
547  bdeck_list = [bdeck_file]
548  adeck_list = []
549  edeck_list = []
550 
551  # get adeck files
552  if self.c_dict['GET_ADECK']:
553  adeck_list = self.find_deck_files('A', current_basin,
554  current_cyclone, model_list,
555  time_info)
556  # get edeck files
557  if self.c_dict['GET_EDECK']:
558  edeck_list = self.find_deck_files('E', current_basin,
559  current_cyclone, model_list,
560  time_info)
561 
562  if not adeck_list and not edeck_list:
563  self.log_error('Could not find any corresponding '
564  'ADECK or EDECK files')
565  continue
566 
567  # reformat extra tropical cyclone files if necessary
568  if self.c_dict['REFORMAT_DECK']:
569  adeck_list = self.reformat_files(adeck_list, 'A', time_info)
570  bdeck_list = self.reformat_files(bdeck_list, 'B', time_info)
571  edeck_list = self.reformat_files(edeck_list, 'E', time_info)
572 
573  self.adeck = adeck_list
574  self.bdeck = bdeck_list
575  self.edeck = edeck_list
576
