203  def getLowestForecastFile(self, valid_time, dtype, template):

207  @param valid_time valid time to search

214  min_forecast = time_util.get_seconds_from_string(self.c_dict[dtype+'_MIN_FORECAST'], 'H')
215  max_forecast = time_util.get_seconds_from_string(self.c_dict[dtype+'_MAX_FORECAST'], 'H')
216  smallest_input_accum = min([lev['amount'] for lev in self.c_dict['ACCUM_DICT_LIST']])
217 
218  # if smallest input accumulation is greater than an hour, search hourly
219  if smallest_input_accum > 3600:
220  smallest_input_accum = 3600
221 
222  min_forecast_string = time_util.ti_get_lead_string(min_forecast)
223  max_forecast_string = time_util.ti_get_lead_string(max_forecast)
224  smallest_input_accum_string = time_util.ti_get_lead_string(smallest_input_accum, plural=False)
225  self.logger.debug(f"Looking for file with lowest forecast lead valid at {valid_time}"
226  f" between {min_forecast_string} and {max_forecast_string} using "
227  f"{smallest_input_accum_string} intervals")
228

231  input_dict = {}

254  def get_daily_file(self, time_info, accum, data_src, file_template):
255  """!Pull accumulation out of file that contains a full day of data
256  Args:
257  @param time_info dictionary containing timing information
258  @param accum accumulation to extract from file
259  @param data_src type of data (FCST or OBS)

261  @rtype bool
262  @return True if file was added to output list, False if not"""
263 
264  data_interval = self.c_dict[data_src + '_DATA_INTERVAL']
265  times_per_file = self.c_dict[data_src + '_TIMES_PER_FILE']
266  search_file = None
267  # loop from valid_time back to data interval * times per file
268  for i in range(0, times_per_file+1):
269  search_time = time_info['valid'] - datetime.timedelta(hours=(i * data_interval))
270  # check if file exists
271  dSts = sts.StringSub(self.logger,
272  file_template,
273  valid=search_time,
274  custom=self.c_dict['CUSTOM_STRING'])
275  search_file = os.path.join(self.input_dir,
276  dSts.do_string_sub())
277  search_file = util.preprocess_file(search_file,
278  self.c_dict[data_src+\
279  '_INPUT_DATATYPE'],
280  self.config)
281  if search_file is not None:
282  break
283 
284  if search_file is None:
285  return False
286 
287  diff = time_info['valid'] - search_time
288 
289  # Specifying integer division // Python 3,
290  # assuming that was the intent in Python 2.
291  lead = int((diff.days * 24) // (data_interval))
292  lead += int((diff).seconds // (data_interval*3600)) - 1
293  search_time_info = { 'valid' : search_time,
294  'custom': self.c_dict['CUSTOM_STRING']}
295

302  self.log_error(f'Accumulation {accum} was not specified in the {data_src}'
303  '_PCP_COMBINE_INPUT_ACCUMS list')
304  return False
305 
306  # if name was not set in the input levels list, use accumulation time in MET time format
307  if fname is None:
308  addon = time_util.time_string_to_met_time(accum, default_unit='S')
309  else:
310  fname = sts.StringSub(self.logger, fname, **search_time_info).do_string_sub()
311  addon = "'name=\"" + fname + "\";"
312 
313  # if name is a python script, don't set level
314  if not util.is_python_script(fname):
315  addon += " level=\"(" + str(lead) + ",*,*)\";"
316 
317  addon += "'"
318 
319  self.add_input_file(search_file, addon)

408  def get_accumulation(self, time_info, accum, data_src):
409  """!Find files to combine to build the desired accumulation
410  Args:
411  @param time_info dictionary containing time information
412  @param accum desired accumulation to build
413  @param data_src type of data (FCST or OBS)
414  @rtype bool
415  @return True if full set of files to build accumulation is found
416  """
417  in_template = self.c_dict[data_src+'_INPUT_TEMPLATE']
418 
419  if self.c_dict[data_src + '_IS_DAILY_FILE'] is True:
420  return self.get_daily_file(time_info, accum, data_src, in_template)
421 
422  search_time = time_info['valid']
423  # last time to search is the output accumulation subtracted from the
424  # valid time, then add back the smallest accumulation that is available
425  # in the input. This is done because data contains an accumulation from
426  # the file/field time backwards in time
427  # If building 6 hour accumulation from 1 hour accumulation files,
428  # last time to process is valid - 6 + 1
429  accum_relative = time_util.get_relativedelta(accum, 'H')
430  # using 1 hour for now
431  smallest_input_accum = min([lev['amount'] for lev in self.c_dict['ACCUM_DICT_LIST']])
432  if smallest_input_accum == 9999999:
433  smallest_input_accum = 3600
434 
435  last_time = time_info['valid'] -\
436  accum_relative +\
437  datetime.timedelta(seconds=smallest_input_accum)
438 
439  total_accum = time_util.ti_get_seconds_from_relativedelta(accum_relative,
440  time_info['valid'])
441 
442  # log the input and output accumulation information
443  search_accum_list = []
444  for lev in self.c_dict['ACCUM_DICT_LIST']:
445  if lev['template'] is not None:
446  search_accum_list.append(lev['template'])
447  else:
448  search_accum_list.append(time_util.ti_get_lead_string(lev['amount'], plural=False))
449 
450  search_accum_string = ' or '.join(search_accum_list)
451  self.logger.debug(f"Trying to build a {time_util.ti_get_lead_string(total_accum, plural=False)} accumulation using {search_accum_string} input data")
452 
453  # loop backwards in time until you have a full set of accum
454  while last_time <= search_time:
455  found = False
456 
457  if total_accum == 0:
458  break
459 
460  # look for biggest accum that fits search
461  for accum_dict in self.c_dict['ACCUM_DICT_LIST']:
462  if accum_dict['amount'] > total_accum and accum_dict['template'] is None:
463  continue
464 
465  search_file, lead = self.find_input_file(in_template, time_info['init'], search_time,
466  accum_dict['amount'], data_src)
467 
468  # if found a file, add it to input list with info
469  if search_file is not None:
470  # if template is used in accum, find value and apply bucket interval is set
471  if accum_dict['template'] is not None:
472  accum_amount = self.get_template_accum(accum_dict,
473  search_time,
474  lead,
475  data_src)
476  if accum_amount > total_accum:
477  self.logger.debug("Accumulation amount is bigger than remaining accumulation.")
478  continue
479  else:
480  accum_amount = accum_dict['amount']
481 
482  accum_met_time = time_util.time_string_to_met_time(accum_amount)
483  addon = self.get_addon(accum_dict, accum_met_time, search_time)
484  # add file to input list and step back in time to find more data
485  self.add_input_file(search_file, addon)
486  self.logger.debug(f"Adding input file: {search_file} with {addon}")
487  search_time = search_time - datetime.timedelta(seconds=accum_amount)
488  total_accum -= accum_amount
489  found = True
490  break
491 
492  # if we don't need any more accumulation, break out of loop and run
493  if not total_accum:
494  break
495 
496  # if we still need to find more accum but we couldn't find it, fail
497  if not found:
498  return False
499 
500  # fail if no files were found or if we didn't find
501  # the entire accumulation needed
502  if not self.infiles or total_accum:
503  return False
504 
505  return True
